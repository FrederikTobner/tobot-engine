358 template <typename... Args>
359 void log(LogPriority log_priority, char const * format, Args... args) {
360 if (this->priority <= log_priority) {
361 std::scoped_lock lock(log_mutex);
362 std::time_t current_time = std::time(0);
363 std::tm * timestamp = std::localtime(&current_time);
364 char buffer[80];
365 strftime(buffer, 80, get_Instance().timeStampFormat, timestamp);
366 std::cout << buffer << " - [";
367#ifdef OS_WINDOWS
368 HANDLE console_color = GetStdHandle(STD_OUTPUT_HANDLE);
369 SetConsoleTextAttribute(console_color, get_Instance().logLevelColors[log_priority]);
370#endif
371 std::cout << logLevelStrings[log_priority];
372#ifdef OS_WINDOWS
373 SetConsoleTextAttribute(console_color, 7);
374#endif
375 if (log_priority == LogPriority::CRITICAL_PRIORITY) {
376 std::cout << "]\t";
377 } else {
378 std::cout << "]\t\t";
379 }
380#ifdef OS_WINDOWS
381 if (log_priority == CRITICAL_PRIORITY) {
382 SetConsoleTextAttribute(console_color, 4);
383 }
384#endif
385 printf(format, args...);
386 std::cout << std::endl;
387#ifdef OS_WINDOWS
388 if (log_priority == CRITICAL_PRIORITY) {
389 SetConsoleTextAttribute(console_color, 7);
390 }
391#endif
392 if (this->file) {
393 fprintf(this->file, "%s - [%s]\t", buffer, logLevelStrings[log_priority]);
394 if (log_priority == LogPriority::CRITICAL_PRIORITY) {
395 fprintf(this->file, "\t");
396 }
397 fprintf(this->file, format, args...);
398 fprintf(this->file, "\n");
399 }
400 }
