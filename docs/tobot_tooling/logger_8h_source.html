369 template <typename... Args>
370 void log(LogPriority log_priority, char const * format, Args... args) {
371 if (this->priority <= log_priority) {
372 std::scoped_lock lock(log_mutex);
373 std::time_t current_time = std::time(0);
374 std::tm * timestamp = std::localtime(&current_time);
375 char buffer[80];
376 strftime(buffer, 80, get_Instance().timeStampFormat, timestamp);
377 printf("%s - [", buffer);
378#ifdef OS_WINDOWS
379 HANDLE console_color = GetStdHandle(STD_OUTPUT_HANDLE);
380 SetConsoleTextAttribute(console_color, get_Instance().logLevelColors[format_priority]);
381#endif
382 std::cout << logLevelStrings[log_priority];
383#ifdef OS_WINDOWS
384 SetConsoleTextAttribute(console_color, 7);
385#endif
386 if (log_priority == LogPriority::CRITICAL_PRIORITY) {
387 std::cout << "]\t";
388 } else {
389 std::cout << "]\t\t";
390 }
391#ifdef OS_WINDOWS
392 if (format_priority == CRITICAL_PRIORITY) {
393 SetConsoleTextAttribute(console_color, 4);
394 }
395#endif
396 printf(format, args...);
397 std::cout << "\n";
398#ifdef OS_WINDOWS
399 if (format_priority == CRITICAL_PRIORITY) {
400 SetConsoleTextAttribute(console_color, 7);
401 }
402#endif
403 if (this->file) {
404 fprintf(this->file, "%s - [%s]\t", buffer, logLevelStrings[log_priority]);
405 if (log_priority == LogPriority::CRITICAL_PRIORITY) {
406 fprintf(this->file, "\t");
407 }
408 fprintf(this->file, format, args...);
409 fprintf(this->file, "\n");
410 }
411 }
