144 typename AVLTree<T>::Node * AVLTree<T>::insert(Node * node, T value) {
145 if (node == nullptr) {

148 if (value < node->value) {
149 node->left = insert(node->left, value);

177 template <typename T>
178 typename AVLTree<T>::Node * AVLTree<T>::remove(Node * node, T value) {
179 if (node == nullptr) {
180 return node;
181 }
182 if (value < node->value) {
183 node->left = remove(node->left, value);
184 } else if (value > node->value) {
185 node->right = remove(node->right, value);
186 } else {
187 if (node->left == nullptr || node->right == nullptr) {
188 Node * temp = node->left ? node->left : node->right;
189 if (temp == nullptr) {
190 temp = node;
191 node = nullptr;
192 } else {
193 *node = *temp;
194 }
195 deleteNode(temp);
196 } else {
197 Node * temp = node->right;
198 while (temp->left != nullptr) {
199 temp = temp->left;
200 }
201 node->value = temp->value;
202 node->right = remove(node->right, temp->value);
203 }
204 }
205 if (node == nullptr) {
206 return node;
207 }
208 node->height = 1 + std::max(getHeight(node->left), getHeight(node->right));
209 int balance = getBalance(node);
210 if (balance > 1 && getBalance(node->left) >= 0) {
211 return rotateRight(node);
212 }
213 if (balance > 1 && getBalance(node->left) < 0) {
214 return rotateLeftRight(node);
215 }
216 if (balance < -1 && getBalance(node->right) <= 0) {
217 return rotateLeft(node);
218 }
219 if (balance < -1 && getBalance(node->right) > 0) {
220 return rotateRightLeft(node);
