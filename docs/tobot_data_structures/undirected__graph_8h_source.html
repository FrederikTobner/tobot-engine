273 template <typename T>
274 void UndirectedGraph<T>::addEdgeToConnectedComponents(T vertex1, T vertex2) {
275 // If the edge connects two vertices that are already in the same connected component, then it has created a
276 // cycle.
277 for (std::vector<T> connected_component : connected_components) {
278 if (std::find(connected_component.begin(), connected_component.end(), vertex1) !=
279 connected_component.end() &&
280 std::find(connected_component.begin(), connected_component.end(), vertex2) !=
281 connected_component.end()) {
282 addCycle(vertex1, vertex2);
283 return;
284 }
285 }
286 // If the edge connects two vertices that are not in the same connected component, then add the edge to the
287 // connected components.
288 bool vertex1_found = false;
289 bool vertex2_found = false;
290 for (std::vector<T> & connected_component : connected_components) {
291 if (std::find(connected_component.begin(), connected_component.end(), vertex1) !=
292 connected_component.end()) {
293 if (std::find(connected_component.begin(), connected_component.end(), vertex2) ==
294 connected_component.end()) {
295 connected_component.push_back(vertex2);
296 }
297 vertex1_found = true;
298 }
299 if (std::find(connected_component.begin(), connected_component.end(), vertex2) !=
300 connected_component.end()) {
301 if (std::find(connected_component.begin(), connected_component.end(), vertex1) ==
302 connected_component.end()) {
303 connected_component.push_back(vertex1);
304 }
305 vertex2_found = true;
306 }
307 }
308 // If the edge connects two vertices that are not in any connected component, then create a new connected
309 // component
310 if (!vertex1_found && !vertex2_found) {
311 std::vector<T> connected_component;
312 connected_component.push_back(vertex1);
313 connected_component.push_back(vertex2);
314 connected_components.push_back(connected_component);
315 }
316 }

614 template <typename T>
615 void UndirectedGraph<T>::findCycle(T vertex, std::vector<T> & visited_vertices, std::vector<T> & cycle) {
616 visited_vertices.push_back(vertex);
617 cycle.push_back(vertex);
618 for (T neighbor : getNeighbors(vertex)) {
619 if (std::find(visited_vertices.begin(), visited_vertices.end(), neighbor) == visited_vertices.end()) {
620 findCycle(neighbor, visited_vertices, cycle);
621 } else if (std::find(cycle.begin(), cycle.end(), neighbor) != cycle.end()) {
622 std::vector<T> cycle_copy = cycle;
623 cycle_copy.erase(cycle_copy.begin(), std::find(cycle_copy.begin(), cycle_copy.end(), neighbor));
624 cycles.push_back(cycle_copy);
625 }
626 }
627 cycle.pop_back();

740 template <typename T>
741 std::vector<T> UndirectedGraph<T>::getNeighbors(T vertex, std::vector<T> vertices, std::vector<T> visited_vertices,
742 std::vector<std::pair<T, T>> edges) const {
743 // Create an empty vector to store the neighbors of the input vertex.
744 std::vector<T> neighbors;
745 // loop through all edges to find the neighbors of the input vertex.
746 for (std::pair<T, T> edge : edges) {
747 // If the first vertex of the edge is equal to the input vertex and the second vertex of the edge is within
748 // the input vertices and not included in the visited vertices, add it to the vector of neighbors.
749 if (edge.GetFirst() == vertex &&
750 std::find(vertices.begin(), vertices.end(), edge.GetSecond()) != vertices.end() &&
751 std::find(visited_vertices.begin(), visited_vertices.end(), edge.GetSecond()) ==
752 visited_vertices.end()) {
753 neighbors.push_back(edge.GetSecond());
754 } // If the second vertex of the edge is equal to the input vertex and the first vertex of the edge is
755 // within
756 // the input vertices and not included in the visited vertices, add it to the vector of neighbors.
757 else if (edge.GetSecond() == vertex &&
758 std::find(vertices.begin(), vertices.end(), edge.GetFirst()) != vertices.end() &&
759 std::find(visited_vertices.begin(), visited_vertices.end(), edge.GetFirst()) ==
760 visited_vertices.end()) {
761 neighbors.push_back(edge.GetFirst());
762 }
763 }
764 // Return the vector of neighbors.
765 return neighbors;

811 void UndirectedGraph<T>::findConnectedComponent(T vertex, std::vector<T> & visited_vertices,
812 std::vector<T> & connected_component) const {
813 // Adds the initial vertex as visited and part of the connected component
