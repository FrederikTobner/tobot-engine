47 std::vector<Token<T>> Lexer<T>::tokenize(std::string source) {
48 std::size_t line = 1;
49 std::size_t column = 1;
50 std::size_t start = 0;
51 std::size_t end = 0;
52 std::vector<Token<T>> tokens;
53 std::string sourceStart = source;
54 std::string lexeme;
55 std::smatch match;
56 while (!this->isAtEnd(source, start)) {
57 // Skipping whitespaces between the tokens
58 this->skipWhitespace(source, start, line, column);
59 sourceStart = source.substr(start);
60 for (TokenizationRule<T> rule : this->rules) {
61 // Only match at the start of the string
62 if (std::regex_search(sourceStart, match, rule.getPattern())) {
63 // The end of the current lexeme is the length of the match
64 end = start + match.length();
65 lexeme = source.substr(start, end - start);
66 // Adding the token
67 tokens.push_back(Token<T>(rule.getType(), lexeme, line, column));
68 // Setting the current start of the next lexeme to the end of the current lexeme
69 start = end;
70 // Increment column by the length of the match
71 column += match.length();
72 break;
73 }
74 }
75 if (match.empty()) {
76 this->reportError("Unexpected character '" + std::string(1, source[start]), line, column);
77 break;
78 }
79 }
80 return tokens;
81 }
