275  auto UndirectedGraph<T>::addEdgeToConnectedComponents(T vertex1, T vertex2) -> void {
276  // If the edge connects two vertices that are already in the same connected component, then it has created a
277  // cycle.
278  for (std::vector<T> connected_component : connected_components) {
279  if (std::find(connected_component.begin(), connected_component.end(), vertex1) !=
280  connected_component.end() &&
281  std::find(connected_component.begin(), connected_component.end(), vertex2) !=
282  connected_component.end()) {
283  addCycle(vertex1, vertex2);
284  return;
285  }
286  }
287  // If the edge connects two vertices that are not in the same connected component, then add the edge to the
288  // connected components.
289  bool vertex1_found = false;
290  bool vertex2_found = false;
291  for (std::vector<T> & connected_component : connected_components) {
292  if (std::find(connected_component.begin(), connected_component.end(), vertex1) !=
293  connected_component.end()) {
294  if (std::find(connected_component.begin(), connected_component.end(), vertex2) ==
295  connected_component.end()) {
296  connected_component.push_back(vertex2);
297  }
298  vertex1_found = true;
299  }
300  if (std::find(connected_component.begin(), connected_component.end(), vertex2) !=
301  connected_component.end()) {
302  if (std::find(connected_component.begin(), connected_component.end(), vertex1) ==
303  connected_component.end()) {
304  connected_component.push_back(vertex1);
305  }
306  vertex2_found = true;
307  }
308  }
309  // If the edge connects two vertices that are not in any connected component, then create a new connected
310  // component
311  if (!vertex1_found && !vertex2_found) {
312  std::vector<T> connected_component;
313  connected_component.push_back(vertex1);
314  connected_component.push_back(vertex2);
315  connected_components.push_back(connected_component);
316  }
317  }
